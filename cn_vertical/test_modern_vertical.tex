% test_modern_vertical.tex
% 测试纯竖排功能（不使用版心/鱼尾）
% 验证 cn_vertical 可以独立使用

\documentclass{article}
\usepackage{geometry}
\geometry{
    paperwidth=400pt,
    paperheight=300pt,
    top=30pt,
    bottom=30pt,
    left=30pt,
    right=30pt
}
\usepackage{fontspec}
\setmainfont{TW-Kai}[
    RawFeature={+vert,+vrt2},
    CharacterWidth=Full
]
\usepackage{cn_vertical}

\pagestyle{empty}

\begin{document}

% 简单竖排测试 - 无版心
\VerticalRTT[
    grid-width=30pt,
    grid-height=25pt,
    border=true,
    n-column=0,  % 禁用版心列
    border-color=blue
]{
這是一個測試文本。

現代中文竖排不需要版心和魚尾裝飾。

只需要基本的網格排版功能。
}

\end{document}
