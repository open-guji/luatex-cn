\ExplSyntaxOn
% 定义一个简单的命令来加载 Lua
\cs_new:Npn \load_ancient_logic:n #1 {
    \lua_now:n {
        % 使用 Lua 自己的路径搜索，不依赖 TeX 繁琐的转义
        local~file = [[#1]]
        local~path = kpse.find_file(file, "lua") or file
        dofile(path)
    }
}

% 执行加载，指向你的 src 文件夹
\load_ancient_logic:n { src/ancient_vert.lua }

\ExplSyntaxOff
