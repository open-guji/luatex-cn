% titlepage-issue28-v3.tex - 复现 issue #28 书名页样式
% 使用 outer-cols 实现不同列宽
\documentclass{ltc-guji}

\pageSetup{
  columns=6,
  lines=20
}

\contentSetup{
  font-size=12pt
}

\begin{document}

% 书名页布局：
% 总共 6 列 (从右到左)
% - 列 1: 日期 (1 列)
% - 列 2-4: 书名 (3 列，大字)
% - 列 5: 出版信息 (1 列)
% - 列 6: 空白

\clearpage
\begin{正文}
% 右侧：日期 (小字，顶部对齐)
\TextBox[n-cols=1, box-align=top, font-size=10pt]{光緒九年仲春}%
% 中间：书名 (大字，居中，占 3 格外部空间)
\TextBox[n-cols=1, outer-cols=3, box-align=center, font-size=32pt]{童蒙必讀書}%
% 左侧：出版信息 (小字，底部对齐)
\TextBox[n-cols=1, box-align=bottom, font-size=10pt]{武昌書局校刊}%
\end{正文}

\end{document}
